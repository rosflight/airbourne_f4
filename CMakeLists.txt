project(airbourne C CXX ASM)

# Project-Wide Settings
include(cmake/ExampleRule.cmake)
include(cmake/ToolChain.cmake)

add_subdirectory(lib)

add_library(airbourne_core
    src/advanced.cpp
    src/backup_sram.cpp
    src/gpio.cpp
    src/led.cpp
    src/mpu6000.cpp
    src/pwm.cpp
    src/spi.cpp
    src/uart.cpp
    src/analog_digital_converter.cpp
    src/battery_monitor.cpp
    src/hmc5883l.cpp
    src/M25P16.cpp
    src/ms4525.cpp
    src/rc_ppm.cpp
    src/system.c
    src/ublox.cpp
    src/analog_pin.cpp
    src/eeprom.cpp
    src/i2c.cpp
    src/mb1242.cpp
    src/ms5611.cpp
    src/rc_sbus.cpp
    src/system_stm32f4xx.c
    src/vcp.cpp)
target_include_directories(airbourne_core PUBLIC include)
target_link_libraries(airbourne_core CMSIS StdPeriphLib usb)

add_subdirectory(examples)
